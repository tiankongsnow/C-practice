#include <stdio.h>
long long Fbnl(long N){
    long a1 = 0;
    long a2 = 1;
    long a3 = N;
    for (int i = 1; i <N; i++){
        a3 = a2 + a1;
        a1= a2;
        a2 = a3;
    }
    return a3;
}
int main(){
    int n;
    while(scanf("%d", &n) != EOF){
        if(n>92){
            break;
        }
        long tmp = Fbnl(n);
        printf("%ld", tmp);
        printf("\n");
    }
}
